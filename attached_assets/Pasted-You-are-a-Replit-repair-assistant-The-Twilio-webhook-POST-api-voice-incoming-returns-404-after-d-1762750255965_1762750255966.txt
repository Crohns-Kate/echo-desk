   import twilio from "twilio";

   export function registerVoice(app) {
     // --- Test Route ---
     app.post("/api/voice/test", async (req, res) => {
       const vr = new twilio.twiml.VoiceResponse();
       vr.say({ voice: "Polly.Olivia-Neural" }, "This is a test Twi M L response. The system is working.");
       res.type("text/xml").send(vr.toString());
     });

     // --- Incoming Call Route ---
     app.post("/api/voice/incoming", async (req, res) => {
       const vr = new twilio.twiml.VoiceResponse();
       vr.say({ voice: "Polly.Olivia-Neural" }, "Hello and welcome to your clinic. How can I help you today?");
       vr.pause({ length: 1 });
       vr.redirect({ method: "POST" }, "/api/voice/handle?route=start");
       res.type("text/xml").send(vr.toString());
     });

     // --- Handle Route ---
     app.post("/api/voice/handle", async (req, res) => {
       const route = (req.query.route || "start").toString();
       const vr = new twilio.twiml.VoiceResponse();

       if (route === "start") {
         const g = vr.gather({
           input: "speech",
           language: "en-AU",
           timeout: 5,
           speechTimeout: "auto",
           action: "/api/voice/handle?route=book-day",
           method: "POST",
         });
         g.say({ voice: "Polly.Olivia-Neural" }, "System ready. Would you like to book an appointment?");
         return res.type("text/xml").send(vr.toString());
       }

       if (route === "book-day") {
         const g = vr.gather({
           input: "speech",
           language: "en-AU",
           timeout: 5,
           speechTimeout: "auto",
           action: "/api/voice/handle?route=book-part",
           method: "POST",
         });
         g.say({ voice: "Polly.Olivia-Neural" }, "Which day suits you best?");
         return res.type("text/xml").send(vr.toString());
       }

       vr.say({ voice: "Polly.Olivia-Neural" }, "Sorry, I didn’t catch that. Let’s start again.");
       vr.redirect({ method: "POST" }, "/api/voice/handle?route=start");
       res.type("text/xml").send(vr.toString());
     });
   }
