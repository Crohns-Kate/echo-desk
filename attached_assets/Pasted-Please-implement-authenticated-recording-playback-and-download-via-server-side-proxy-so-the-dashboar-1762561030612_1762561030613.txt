        const recordingUrl = `https://api.twilio.com/2010-04-01/Accounts/${process.env.TWILIO_ACCOUNT_SID}/Recordings/${sid}.mp3`;

     d) For /download: set Content-Type to application/octet-stream AND
        Content-Disposition: attachment; filename="${sid}.mp3" (or .wav), then pipe to res.
